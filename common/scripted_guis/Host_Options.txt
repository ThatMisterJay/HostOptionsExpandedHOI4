scripted_gui = {

	host_options_expanded_options = {
		context_type = player_context

		window_name = "host_options_container"

		visible = {
			ROOT = { TAG = global_event_target:game_host }
			check_variable = { show_host_options > 0.0 }
		}

		effects = {
			exhaustion_PP_payment_click = {
				add_political_power = -200
				subtract_from_variable = { war_exhaustion_variable = 150 }
			}
			exhaustion_stability_payment_click = {
				add_stability = -0.05
				subtract_from_variable = { war_exhaustion_variable = 200 }
			}
		}

		triggers = {
			exhaustion_PP_payment_click_enabled = {
				check_variable = { war_exhaustion_variable > 149.99 }
				has_political_power > 150
			}
			exhaustion_stability_payment_click_enabled = {
				check_variable = { war_exhaustion_variable > 199.99 }
				has_stability > 0.05
			}
		}

		effects = {
			close_button_click = {
				set_variable = { show_host_options = 0 }
			}

			player_flag_click = {
				capital_scope = {
					ROOT = {
						goto_state = PREV
					}
				}
			}

			host_flag_click = {
				event_target:game_host = {
					capital_scope = {
						ROOT = {
							goto_state = PREV
						}
					}
				}
			}
		}

		dynamic_lists = {
			player_list_gridbox = {
				array = global.playerbase
				entry_container = player_list_entry
				change_scope = yes
			}
			fascism_player_list_gridbox = {
				array = global.fascist_countries_array
				entry_container = fascism_player_list_entry
				change_scope = yes
			}
		}

		properties = {
			host_flag = {
				image = "[game_host.GetFlag]"
			}
			player_flag = {
				image = "[This.GetFlag]"
			}
		}
	}
}
